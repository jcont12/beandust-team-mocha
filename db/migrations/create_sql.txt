CREATE TABLE Orders(
   id SERIAL NOT NULL PRIMARY KEY,
   user_id INTEGER,
   item VARCHAR(255),
   quantity INTEGER,
   description VARCHAR(255),
   confirmation_number VARCHAR(255),
   price VARCHAR(255),
   createdAt DATE NOT NULL,
   updatedAt DATE NOT NULL
);
